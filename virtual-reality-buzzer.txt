# Input variables
# X_RADIO_RX, -999 to 999

# Output variables
# P_BUZZER, 0 to 100

# Gist:
# DAT will be last seen value
# If X_RADIO_RX === -999, go to BUZZ
# Save X_RADIO_RX to DAT

# BUZZ:
# If DAT === 1, NOT ACC
# If DAT === 0, ACC = 0
# Send ACC to P_BUZZER

# Controller 1 (MC6000):
teq X_RADIO_RX -999
+ jmp BUZZ
# DEV: Will `X_RADIO_RX` be flushed by now?
mov X_RADIO_RX dat

BUZZ
