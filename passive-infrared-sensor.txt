# Input variables
# rtc, 0 to 100
# sensor, 0 to 100
# on time, -999 to 999
# off time, -999 to 999

# Output variables
# alarm, 0 to 100

# Gist:
# Controller 1:
#   Normalize SENSOR into _ALARM (i.e. 0-100 -> 0/100)
# Controller 2:
#   When _ALARM, after ON TIME, and before OFF TIME, send ALARM

# Controller 1
# If sensor is at least 20
tgt SENSOR 20
  # Send alarm signal to controller 2
+ mov 100 _ALARM
  # Otherwise, reset alarm signal for controller 2
- mov 0 _ALARM
