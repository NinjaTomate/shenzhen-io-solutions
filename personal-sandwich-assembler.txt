so swapping to ram and this is an interim state

Controller 1 MC6000

```
START:
# Wait for signal
slx x1

# Save input
mov x1 dat

# If we should start then send start signal
teq dat 3
+ mov 100 x2
# DEV: Jump to prevent touching memory
+ jmp START

# If we are at -1
teq dat -1
# Move to first option offset
+ mov 1 x0
# Set it to 0 (RAM jumps to next option -- i.e. 2)
+ mov 0 x3
# Set it to 0 also
+ mov 0 x3
# Otherwise, jump to option offset and set it
# DEV: Security flaw since it's outside input \o/
- mov dat x0
- mov 1 x3
```

Controller 2 MC6000

# Wait for start signal
slx x1
# Flush signal
mov x1 null
# Drop bread + sleep
gen p1 1 0
# Drop meat + sleep
mov 100 x2
slp 1
# Access cheese RAM
mov 1 x0
# If cheese (i.e. not "no cheese"), drop it + sleep
teq x3 1
- mov 010 x2
- slp 1
- mov 000 x2
# Drop mustard
gen p0 1 0
# Mustard RAM already accessed
# If extra mustard, drop it + sleep
teq x3 1
+ gen p0 1 0
# Drop bread + sleep
gen p1 1 0
